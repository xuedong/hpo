[docs]def evaluate_random(bo_model, loss, n_eval=20):
    res = []
    for i in range(n_eval):
        param = bo_model.sample_param()
        current_loss = loss(**param)
        res.append(current_loss)
        print('Param {}, Loss: {}'.format(param, current_loss))
    return res
